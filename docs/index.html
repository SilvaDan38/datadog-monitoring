# Python — span customizado com tags de negócio
from ddtrace import tracer

@app.post("/products")
def create_product(payload: ProductCreate, db = Depends(get_db)):
    with tracer.trace("products.create", resource="POST /products") as span:
        span.set_tag("product.name",  payload.name)
        span.set_tag("product.price", payload.price)
        span.set_tag("product.category", payload.category)

        product = Product(**payload.dict())
        db.add(product)
        db.commit()

        span.set_tag("product.id", product.id)  # ID gerado
        return product

# Execução com tracer:
# DD_SERVICE=python-crud DD_ENV=local ddtrace-run uvicorn main:app
// Node.js — inicialização e span customizado
const tracer = require('dd-trace').init({
  service:    'nodejs-crud',
  env:        process.env.DD_ENV || 'local',
  logInjection:    true,   // correlaciona logs ↔ traces
  profiling:       true,
  dbmPropagationMode: 'full',  // DBM
  appsec: { enabled: true }    // AppSec
});

app.post('/products', async (req, res) => {
  const span = tracer.startSpan('products.create');
  try {
    const product = await Product.create(req.body);
    span.setTag('product.id', product.id);
    res.status(201).json(product);
  } finally {
    span.finish();
  }
});

// Execução:
// node -r dd-trace/init server.js
// Java — @Trace annotation + span manual
import datadog.trace.api.Trace;
import datadog.trace.api.GlobalTracer;

@PostMapping("/products")
@Trace(operationName = "products.create",
       resourceName  = "POST /products")
public ResponseEntity<Product> createProduct(@RequestBody Product p) {
    Tracer tracer = GlobalTracer.get();
    Span span = tracer.activeSpan();

    if (span != null) {
        span.setTag("product.name",  p.getName());
        span.setTag("product.price", p.getPrice());
    }

    Product saved = repository.save(p);
    return ResponseEntity.status(HttpStatus.CREATED).body(saved);
}

// Execução:
// java -javaagent:dd-java-agent.jar \
//      -Ddd.service=java-crud \
//      -jar app.jar
// .NET — Minimal API com span customizado
using Datadog.Trace;

app.MapPost("/products", async (ProductCreateDto dto, AppDbContext db) => {
    using var scope = Tracer.Instance.StartActive("products.create");

    scope.Span.SetTag("product.name",  dto.Name);
    scope.Span.SetTag("product.price", dto.Price);

    var product = new Product { Name = dto.Name, Price = dto.Price };
    db.Products.Add(product);
    await db.SaveChangesAsync();

    scope.Span.SetTag("product.id", product.Id);
    return Results.Created($"/products/{product.Id}", product);
});

// Auto-instrumentação via variáveis de ambiente:
// CORECLR_ENABLE_PROFILING=1
// CORECLR_PROFILER={846F5F1C-F9AE-4B07-969E-05C26BC060D8}
# Ruby — Datadog::Tracing.trace
require 'ddtrace'

post '/products' do
  Datadog::Tracing.trace('products.create',
                          resource: 'POST /products') do |span|

    body_data = JSON.parse(request.body.read)
    product = Product.new(body_data.slice('name', 'price', 'category'))

    unless product.save
      halt 422, json(errors: product.errors.full_messages)
    end

    span.set_tag('product.id',   product.id)
    span.set_tag('product.name', product.name)

    status 201
    json product.as_json
  end
end

# Execução:
# ddtracerb exec ruby app.rb
